۸۳ - سورة المطففين
وَيْـل ( ۱ )
عَذابٌ أوْ هَلاك أو وادٍ في جهـنـّـم
للمطـفـفين ( ۱ )
المُـنقـِـصين في الكـيْـل ، و مثله الوزن
اكتالوا ( ۲ )
اشْتروْا بالكيل ، و مثله الوزن
كالوهمْ ( ۳ )
أعطَوْا غـيرهم بالكيل
وَزنوهمْ ( ۳ )
أعْطَوْا غـيرهم
يُـخْـسِرون ( ۳ )
يَنـقـصون الكيل و الوَزن
لربّ العالمين ( ٦ )
لأمره و حكمه
كَـتاب الفجّـار ( ۷ )
ما يُـكتب من أعمالهم
لفي سجين ( ۷ )
لمُـثـبَـتٌ في ديوان الشّر
كتابٌ مَرقوم ( ۹ )
بَـيّـن الكتابة أو معلـّـم بعلامة
مُعْـتدٍ ( ۱۲ )
فاجر مُتجاوز عن نـَـهْج الحقّ
أساطير الأوّلين ( ۱۳ )
أباطيلهم المسَـــّطرة في كتبهم
كلاّ ( ۱٤ )
رَدْعٌ و زجْرٌ عن قولهم الباطل
رانَ على قـلوبهم ( ۱٤ )
غَـلبَ و غَـــّطى عليها أو طبع عليها
لصالوا الجحيم ( ۱٦ )
لداخِلوها أو لمقاسو حَرّها
كتاب الأبرار ( ۱۸ )
ما يُكتب من أعمالهم
لفي عليين ( ۱۸ )
لمُثبت في ديوان الخيْر
الأرائك ( ۲۳ )
الأسِرّة في الحجال(جمع حجلة محركة – بيت يزيّن بالقباب و الأسرة و الستور)
نضْرة النّعيم ( ۲٤ )
بهجته و رونقه و بَهاءَه
رَحيق ( ۲۵ )
أجْوَد الخَمْر و أصْـفاه
مَختوم ( ۲۵ )
إناؤه حتى يَفـكّه الأبرار
خِـتامه مسْـك ( ۲٦ )
ختام إنائه المِسْـك بَدَل الــّطين
فلـيـتنافس ( ۲٦ )
فـلـيَـتـَـسارعْ . أو فـلـيَسْـتـَـبـِـقْ
مِزَاجُه ( ۲۷ )
ما يُـمْـزَجَ بهِ و يُـخْـلـَـط
تسْـنيم ( ۲۷ )
عَـيْـن ٍ عاليةٍ شَرابُـها أشرف شراب
يَشرَب بها ( ۲۸ )
يَشرَب منها
يَغامَـزون ( ۳۰ )
يُشيرون إليهم بلأعين استهزاء
فـكهين ( ۳۱ )
مُـتـلذذين باستِخفافهم بالمؤمنين
ثـُـوّب الكـفار ( ۳٦ )
جُوزُوا بسُخريّـتهم بالمؤمنين
