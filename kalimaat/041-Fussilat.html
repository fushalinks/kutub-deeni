٤۱ - سورة فصلت
( ۳ ) فصّـلتْ آياته :مُـيّـزَتْ و نوّعَتْ. أو بُـيّـنتْ
( ۵ ) أكنّة :أغطية خِلقـيّة تمنع الفَهم
( ۵ ) وَقـْـرٌ :صَممٌ و ثقـلٌ يمنعُ السّمع
( ۵ ) حِجاب :سِتـْـرٌ غليظ يمنع التـّواصل
( ٦ ) فاستقيموا إليه :تـَوجّهوا إليه بطاعته و عبادته
( ٦ ) وَيْـل للمشركين :هلاكٌ أو حَسْرَةٌ أو شدة عذاب لهم
( ۸ ) غير ممنون :غير مقطوع عنهم
( ۹ ) أندادًا :أمْثالا مِنَ مَخـْـلوقاته تـَعبدونها
( ۱۰ ) رَواسيَ :جبالاً ثوابتَ تمنعُها المَـيَـدَان
( ۱۰ ) بارك فيها :كثـّرَ خَيْرها و مَنافعها
( ۱۰ ) أقواتها :أرزاق أهلها و ما يصلح لمعايشهمْ
( ۱۰ ) في أربعة أيام :في تتمّة أربعة أيّـام
( ۱۰ ) سَوَاءً :اسْتوت الأربعة اسْتِواءً (تـمّتْ)
( ۱۱ ) استوى :عَمَدَ و قصَدَ قصدا سويّـا . .
( ۱۱ ) هي دُخان :مُكونة مما يُشبه الدّخان
( ۱۱ ) ائتيا :افعَلا ما أمرتكما به و جيئا به
( ۱۲ ) فقضاهنّ :أحْكَمَ و أبْدَعَ خلقهنّ
( ۱۲ ) أوْحى :كوّن ، أو دَبّرَ في اليومين
( ۱۲ ) حفظا :حفِظناها حِفظا من الآفات
( ۱۳ ) أنذرْتكم صاعقة :خوّفتكمْ عَذاباً شديدا مهلكا
( ۱٦ ) ريحا صرصرا :شديدة السّموم ، أو البَرْد، أو الصّوت
( ۱٦ ) أيّـام نحِسات :مشـْـئوماتٍ ، أو ذوات غبار و تراب
( ۱٦ ) أخزَى :أشدّ إذلالاً و إهانة
( ۱۷ ) فهديْناهمْ :بيّـنّـا لهم طريقي الضلالة و الهُدَى
( ۱۷ ) العَذاب الهون :المهين
( ۱۹ ) فهم يوزعون :يُـحـْـبَـسُ سَوَابقهم ليلحقهم تواليهم
( ۲۲ ) تستترون :تستخفونَ عند ارتكابكم الفواحش
( ۲۲ ) أنْ يشهد . . :مَخافة أن يَشهد . .
( ۲۲ ) ظنـَـنـْـتمْ :اعتقدْتم عند استتاركم من الناس
( ۲۲ ) كثيرا مما تعملون :و هو ما عمِلتم خِفْية
( ۲۳ ) أرداكمْ :أهْـلكَكُمْ
( ۲٤ ) مَثوًى لهمْ :مَحَلّ ثوّاء و إقامة أبديّة لهم
( ۲٤ ) إنْ يَستعتبوا :يَطلبوا رضاءَ ربهم يومئذ
( ۲٤ ) من المعتبينَ :منَ المجَابين إلى ما طَلبوا
( ۲۵ ) قيّضنا لَهمْ :سبّـبنا و هَـيّـأنا لهمْ
( ۲۵ ) حقّ عليهم القول :وَجَبَ و ثـَبَـتَ عليهم وَعيد العذاب
( ۲٦ ) الغوْا فيه :ائـْتوا باللّغو و الباطل عند قراءته
( ۲۹ ) الأسْـفـلين :في الدّرك الأسفل من النار
( ۳۰ ) استقاموا :على الحقّ اعتقادا و عملا و إخلاصا
( ۳۱ ) ما تدّعون :ما تتمنّونه و تطلبونه
( ۳۲ ) نزلاً :رزقا أو ضِـيافةو تكرمة ، أو منّـا
( ۳٤ ) وليّ حميمٌ :صَديقٌ قريبٌ يَهتمّ لأمركَ
( ۳۵ ) ما يلقـّـاها :ما يُؤتـَى هذه الخصلة الشريفة
( ۳٦ ) ينزغـنـّـك :يُصيبنّـك. أو يَصرفـنّـك
( ۳٦ ) نزْغٌ :وَسْوَسة . أو صارفٌ
( ۳۸ ) لا يسأمون :لا يَمَـلـّـون التـّـسبيح
( ۳۹ ) الأرض خاشعة :يَابسة مُـتطامِنـَـة جَدْبَة
( ۳۹ ) اهتزّت :تحرّكتْ بالنّبات
( ۳۹ ) رَبَتْ :انتفختْ و عَلَتْ
( ٤۰ ) يُـلْحِدونَ :يَميلون عن الحقّ و الاستقامة
( ٤۱ ) إنّ الذين كَـفَـروا :خَبَر "إنّ" تقديره "لا يَخْـفَوْنَ علينا" أو "هالِكون
( ٤٤ ) قرآنا أعجميـّـا :بـِـلغَة العجَم كما اقترحوا
( ٤٤ ) لولا فصّـلت آياته :هلاّ بُـيّـنـَـتْ آياته بلسان نعرفه
( ٤٤ ) أأعجميّ و عربيّ :أقرآنٌ أعجميّ و رَسولٌ عرَبيّ
( ٤٤ ) في آذانهم وَقـْـرٌ :صَمَمٌ مانعٌ من سماعه
( ٤٤ ) هو عليهم عمًى :ظلمَة و شبْـهَة مُسْتولِـيَة عليهم
( ٤۵ ) مريب :مُوقِع في الرّيبة و القلق
( ٤۷ ) أكمامِها :أوْعِـيَـتها
( ٤۷ ) آذنـّاك :أخبَرناك و أعْـلمناك
( ٤۸ ) ظنـّوا :أيقـنوا
( ٤۸ ) محيص :مَهرب و مفرّ من العذاب
( ٤۹ ) لا يسأم الإنسان :لا يملّ و لا يفـتر
( ٤۹ ) دعاء الخير :طَلبه العافية و السّعَة في النّعمة
( ٤۹ ) فَـيَـئوس قنوط :من فضل الله و رحمته
( ۵۰ ) هذا لي :هذا حقـّـي أستحقـّـه بعملي
( ۵۰ ) عذاب غليظ :شديد لا يُـفـَـتـّـرُ عنهم
( ۵۱ ) نأى بجانبه :تباعد عن الشكر بكـلـّـيَـتِه تكَـبّرًا
( ۵۱ ) دُعَاءٍ عريض :كثيرٍ مسْـتـَـمرّ
( ۵۲ ) أرأيتمْ :أخبروني
( ۵۳ ) الآفاق :أقطار السّموات و الأرض
( ۵٤ ) مِرْية :شكّ ٍ عَظيم
