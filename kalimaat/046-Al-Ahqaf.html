٤٦ - سورة الأحقاف
أجل مسمّى ( ۳ )
بتقدير أجل مسمّى و هو يوم القيامة
أرأيتم ( ٤ )
أخبروني
لهم شرك ( ٤ )
شركة و نـَـصيبٌ مع الله تعالى
أثارةٍ من علم ( ٤ )
بَقيّة من علم عندكم
تفيضون فيه ( ۸ )
تندفعون فيه طعنا و تكذيبا
بدْعًا ( ۹ )
بديعا منفردًا فيما جئتُ به
أرأيتم ( ۱۰ )
أخبروني ماذا حالكم
إفكٌ قديم ( ۱۱ )
كذبٌ متقادمٌ
وصّينا الإنسان ( ۱۵ )
أمرناه و ألزمناه
كُرْها ( ۱۵ )
ذات كرْهٍ و مشقة
حمله و فِصاله ( ۱۵ )
مدّة حمله و فطامه من الرّضاع
بَلغ أشدّه ( ۱۵ )
بَلـَغ َ كمال قوته و عقله
ربّي أوزعني ( ۱۵ )
ألهمني ووفقني و رغّبني
أفّ لكما ( ۱۷ )
كلمة تضجّرٍ و تبرّمٍ و كراهيّة
أن أخرَج ( ۱۷ )
أُبْعَثَ من القبر بعد الموت
خلـَـتْ القرون ( ۱۷ )
مضتِ الأمم و لم تبعَث
ويلكَ ( ۱۷ )
هلكتَ و المراد حثـّه على الإيمان
آمنْ ( ۱۷ )
صدّقْ بالله و بالبعْث
أساطير الأوّلين ( ۱۷ )
أباطيلهم المسطّرة في كتبهم
حقّ عليهم القول ( ۱۸ )
وَجَبَ عليهم وعيد العذاب
قد خَلـَتْ ( ۱۸ )
مَضتْ و تقدّمت
عذاب الهون ( ۲۰ )
الهوان و الذلّ
أخا عاد ( ۲۱ )
هودا عليه السّلام
بالأحقاف ( ۲۱ )
وادٍ بين عُمان و أرض مَهْرَة
لتأفكنا ( ۲۲ )
لتصرفنا . أو لتزيلنا بالإفك
عارضا ( ۲٤ )
سحابا يعرض في الأفق
تـُدمّر ( ۲۵ )
تـُهلِك
مكّـنّاهم ( ۲٦ )
أقدرناهم و بسطنا لهم
فيما إن مكّـنّـاكم فيه ( ۲٦ )
في الذي ما مكّـنّـاكم فيه
فما أغنى عنهم ( ۲٦ )
فما دفع عنهم
حاق بهم ( ۲٦ )
أحاط أو نزل بهم
صرّفنا الآيات ( ۲۷ )
كررناها بأساليب مُختلفة
قربانـًا آلهة ً ( ۲۸ )
متقرّبـًـا بهم إلى الله
إفكُهُمْ ( ۲۸ )
أثر كذبهم في اتخاذها آلهة
يفترون ( ۲۸ )
يختلقونه في قوْلهم إنها آلهة
صرفنا إليك ( ۲۹ )
أمَلْـنا ووجهنا نَحْوَك
أنصتوا ( ۲۹ )
اسكتوا و اصغوا لنسمعه
قضِيَ ( ۲۹ )
أتمّ و فُــِرغ من قراءة القرآن
فليس بمعجزٍ ( ۳۲ )
لله فائت منه بالهَرَب
لم يعيَ بخلقِهنّ ( ۳۳ )
لم يتعب به أو لم يعجز عنه
بلى ( ۳۳ )
هو قادر على إحياء الموتى
أولوا العزم ( ۳۵ )
ذوو الجـِــدّ و الثبات و الصّبر
بلاغ ( ۳۵ )
