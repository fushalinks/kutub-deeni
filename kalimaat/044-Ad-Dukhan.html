٤٤ - سورة الدخان
ليلة مباركة ( ۳ )
ليلة القدر من شهر رمضان
فيها يُفرق ( ٤ )
يفصّل و يُـبَـيّـن
أمرٍحكيم ( ٤ )
محْكمٍ مبرم أو ملتبسٍ بالحكمة
فارتقبْ ( ۱۰ )
انتظر بهؤلاء الشاكّين
بدخان ( ۱۰ )
كناية عن إصابتهم بالجَدْب و المجاعة
يغشى الناس ( ۱۱ )
يشملهم و يحيط بهم
أنّى لهم الذكرى؟ ( ۱۳ )
كيف يتذكّرون و يتـّـعِظون ؟
مُعلّم ( ۱٤ )
يعلـّـمه بشر
يوم نبطش ( ۱٦ )
يوم نأخذ بشدّة و عنف (يوم بدر أو يوم القيامة)
فتنّا ( ۱۷ )
ابتلينا و امتحنّا
أدّوا إليّ عباد الله ( ۱۸ )
سلـّـموا إلي بني إسرائيل
لا تعلوا ( ۱۹ )
لا تتكبّروا . أو لا تفتروا
بسلطان ( ۱۹ )
حجّة و برهان على صدقي
إنّي عذت بربّي ( ۲۰ )
استجرت به و التجَأت إليه
ترجمونِ ( ۲۰ )
تؤذوني. أو تقتلوني بالحجارة
فأسر بعبادي ليلا ( ۲۳ )
سِـرْ ليلا ببني إسرائيل
إنكم متبعون ( ۲۳ )
يتبعكم فرعون و جنوده
البحْرَ رهوا ( ۲٤ )
ساكنـًـا . أو منفرجا مفتوحا
جندٌ ( ۲٤ )
جماعة
نعمَةٍ ( ۲۷ )
تنعّم أو نضارة عيشٍ و لذاذتِه
فاكهين ( ۲۷ )
ناعمين متفكّهين
منظرين ( ۲۹ )
ممهلين بالعّذاب إلى وقت آخر
كان عاليا ( ۳۱ )
متكبّرا جبّارا
العالمين ( ۳۲ )
عالمي زمانهم
فيه بلاءٌ مبين ( ۳۳ )
اختبار ظاهر أو نعمة ظاهرة
بمُنشرين ( ۳۵ )
بمبعوثين بعد موتتنا
قوم تبّع ( ۳۷ )
أبـِـي كَــِرب الحِمْـيَـــِريّ ملك اليمن
يوم الفصْل ( ٤۰ )
يوم القيامة و الحساب
لا يغني مولى . . ( ٤۱ )
لا يدفع قريب. و لا صديق
شجرة الزقوم ( ٤۳ )
من أخبث الشّجر تنبت في النّار
كالمهل ( ٤۵ )
دُرْديّ الزّيت . أو المعدن المُذاب
الحميم ( ٤٦ )
الماء البالغ غاية الحرارة
فاعتلوه ( ٤۷ )
فجرّوه بعنف و قهر
سواء الجحيم ( ٤۷ )
وَسَط النّار
به تمترون ( ۵۰ )
فيه تجادلون و تمارون
سندس ( ۵۳ )
رقيق الدّيباج
إستبرق ( ۵۳ )
غليظه
زوّجناهم بحور عين ( ۵٤ )
قرنّاهم بنساءٍ بيضٍ مخلوقات في الجنّة واسعات الأعين حِسَانُها
يدعون فيها ( ۵۵ )
يطلبون فيها
فارتقب ( ۵۹ )
فانتظر ما يحلّ بهم
