۸٤ - سورة الإنشقاق
السماء انشقّـت ( ۱ )
انصَدعت عند قيام القيامة
أذنت لربها ( ۲ )
اسْـتـمَـعَتْ و انـْـقادَتْ له تعالى
حُـقـّـتْ ( ۲ )
حَقّ الله عليها الاستماع و الانقـياد
الأرْض مُدّتْ ( ۳ )
بُـسِطتْ و سُـوّيَتْ كمَـدّ الأديم
ألـْـقـَت ما فـيها ( ٤ )
لـَـفـَـظتْ ما في جَوْفِها منَ الموتى
تخلـّـتْ ( ٤ )
خَـلـت عنه غاية الخُـلـُـوّ
كادِح إلى ربّـك ( ٦ )
جاهدٌ في عملك إلى لقاء ربّـك
فـمُلاقـيه ( ٦ )
فمُـلاقٍ لا محالة جزاء عملك
يَدْعو ثبورا ( ۱۱ )
يُنادي هَـلاكًـا قائلا يَاثـُـبُـراه
يَصلى سعيرا ( ۱۲ )
يدْخلها أو يُـقاسي حرّها
لنْ يَحور ( ۱٤ )
لن يرْجع إلى ربّـه تكْذيبا بالبَعْث
فلا أقسِم ( ۱٦ )
أقسِم و "لا" مزيدة
بالشّـفـق ( ۱٦ )
بالحُمْرَة في الأفق بعد الغروب
ما وَسَق ( ۱۷ )
ما صَمّ و جَمَـعَ ما انتشر بالنـّهار
اتـّـسق ( ۱۸ )
اجْـتـمَعَ و تكامل و تمّ نوره
لتـركبنّ ( ۱۹ )
لتـُـلاقُـنّ أيّها النـّاس ( جواب القسم)
طبَـقـًـا ( ۱۹ )
أحْوَالا بَعْدَ أحْوَال مُتطابقة في الشّدة
يوعـون ( ۲۳ )
يُضمِرونه أو يَجمعونه من السّيئات
غيْرُ مَمْـنون ( ۲۵ )
غير مقطوع عنهم
