۵۹ - سورة الحشر
سبّـح لله ( ۱ )
نزّهَه و مَجّـدَهُ تعالى و دلّ عليه
الذين كفروا ( ۲ )
هو يهود بني النضير قرب المدينة
لأوّل الحَشر ( ۲ )
في أول إخراج و إجلاء إلى الشام
فأتاهم الله ( ۲ )
فأتاهم أمره و عقابه
لمْ يحتسِبوا ( ۲ )
لم يظنوا و لم يخطر لهم ببَال
قذف ( ۲ )
ألقى و أنزل إنزالا شديدا
الجَلاء ( ۳ )
الخروج من الوَطن
شاقّوا ( ٤ )
عادَوْا و عَصَوْا و حادّوا
لِـينة ( ۵ )
نخلة . أو نخلة كريمة
على أصولها ( ۵ )
على سوقها
و ما أفاء الله ( ٦ )
و ما ردّ و ما أعادَ
فما أوْجَفتم عليه ( ٦ )
فما أجْرَيتم على تحصيله
ركاب ( ٦ )
ما يُركب ن الإبل خاصّة
دُولة بين الأغنياء ( ۷ )
مِلكا مُتداولا بينهم خاصة
تـبَـوّءوا الدار و الإيمام ( ۹ )
تـَـوَطّـنوا المدينة و أخلصوا الإيمان
حاجَة ( ۹ )
حَزَازَةً و حَسَدًا
خصاصة ( ۹ )
فقـْـرٌ و احتياج
منْ يوق ( ۹ )
منْ يُجَنّبْ و يُـكْـفَ
شحّ نفسه ( ۹ )
بُخْـلها مع الحرص على المنع
غِلاّ ( ۱۰ )
حِقـدًا و بُـغـضـا و غِـشّـا
بَأسهم بينهم ( ۱٤ )
قتالهم فيما بينهم
قلوبهم شتـّـى ( ۱٤ )
متفرقة لتـَـعَاديهمْ
وَبال أمرهم ( ۱۵ )
سوء عاقبة كفرهم
نسوا الله ( ۱۹ )
لمْ يراعوا أوامره و نـَـواهيه
فأنـْـساهمْ أنفسهم ( ۱۹ )
فلم يُقـدّموا لها ما ينفعها عنده
خاشِعًا ( ۲۱ )
ذليلا خاضعا
متصدّعًا ( ۲۱ )
مُـتـشـقـّــقـا
المَلِك ( ۲۳ )
المالك لكل شيء المتصرف فيه
القدوس ( ۲۳ )
البَليغ في النـّـزاهة عن الـنّـقائص
السلام ( ۲۳ )
ذو السّـلامة منْ كل عيّب و نقـْص
المؤمن ( ۲۳ )
المصَدّق لرسلِه بالمعجزات
المهيمن ( ۲۳ )
الرقيب على كلّ شيء
العزيز ( ۲۳ )
القوّي الغالب
الجبّار ( ۲۳ )
القهار . أو العظيم
المتكبّر ( ۲۳ )
البَليغ الكبرياء و العظمة
البارئ ( ۲٤ )
المبدع المخترع
المصوّر ( ۲٤ )
خالق الصّور على ما يريد
الأسماء الحسنى ( ۲٤ )
الدّالة على محاسن المعاني
