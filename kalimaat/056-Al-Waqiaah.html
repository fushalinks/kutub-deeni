۵٦ - سورة الواقعة
وقعت الواقعة ( ۱ )
قامت القيامة بنفخة البعْث
كاذبة ( ۲ )
نفْسٌ كاذبة تـُـنـْـكر وقوعها
خافضة رافعة ( ۳ )
هي خافضة للأشقياء رافعة للسّعداء
رُجّت الأرض ( ٤ )
زلزلتْ وحُرّكت تحريكا بشدة
بُسّت الجبال ( ۵ )
فـتـّـتْ كالسّويق المَـلْـتوت
هباءً منبثـّـا ( ٦ )
غبارًا متفرقا منتشِرا
كنتم أزواجا ( ۷ )
أصنافا
فأصحاب الميمنة ( ۸ )
اليُـمْن و البركة . أو ناحية اليمين
أصحاب المشأمة ( ۹ )
الشؤم . أو ناحية الشمال
ثـلـّـة ( ۱۳ )
هم أمّة من الناس كثيرة
سُرُر موضونة ( ۱۵ )
منسوجة من الذهب بإحكام
ولدان مُخلّدون ( ۱۷ )
مُبْـقون على هيئة الولدان في البهاء
بأكواب ( ۱۸ )
أقداح لا عُرى لها و لا خراطيم
أباريق ( ۱۸ )
أوان لها عُرى و خراطيم
كأس ( ۱۸ )
خمْر أو قدَح فيه خمْر
من معين ( ۱۸ )
خمْر جارية من العيون
لا يُصدّعون عنها ( ۱۹ )
لا يُصيبهم صداع بشربها
لا يُنزفون ( ۱۹ )
لا تذهَبُ عقولهم بسببها
حورٌ عين ( ۲۲ )
نساء بيض واسعات الأعين حِسانُها
اللؤلؤ المكنون ( ۲۳ )
المصون في أصدافه مما يغيّره
لغْوًا ( ۲۵ )
كلاما لا خير فيه أو باطلا
و لا تأثيما ( ۲۵ )
ولا نسبة إلى الإثم أو لا ما يوجبه
في سِدْر ( ۲۸ )
في شجر النـّـبَـق ينعّمون به
مخضود ( ۲۸ )
مقطوع شوكه
طلح ( ۲۹ )
شجر الموز أو مثله
منضود ( ۲۹ )
نضّد بالحمْل من أسفله إلى أعلاه
ظل ممدود ( ۳۰ )
دائم لا يتقلّص أو ممتدّ منبسط
ماءٍ مسكوب ( ۳۱ )
مصبوب يجري في غير أخاديد
مرفوعة ( ۳٤ )
على الأسرّة أو منضدّة مرتفعة
عُرُبا ( ۳۷ )
متحبّبات إلى أزواجهنّ
أترابا ( ۳۷ )
مسْتويات في السّنّ
سموم ( ٤۲ )
ريح شديدة الحرارة تدخل المَسام
حميم ( ٤۲ )
ماءٍ بالغ غاية الحرارة
يَحْموم ( ٤۳ )
دخان شديد السّواد أو نار
لا كريم ( ٤٤ )
لا نافع من أذى الحرّ
مُترَفين ( ٤۵ )
منعّمين متّبعين أهواء أنفسهم
الحِنث ( ٤٦ )
الذنب العظيم – الشرك
زقوم ( ۵۲ )
شجر كريه جدّا في النار
شرب الهيم ( ۵۵ )
الإبل العِطاش التي لا تروى
هذا نزلهم ( ۵٦ )
ما أعدّ لهم من الجزاء
يوم الدين ( ۵٦ )
يوم الجزاء ( يوم القيامة)
أفرأيتم ( ۵۸ )
أخبروني
ما تـُـمْـنون ( ۵۸ )
المنيّ الذي تقذفونه في الأرحام
تخلقونه ( ۵۹ )
تصوّرونه بشرًا سويّا
بمسبوقين ( ٦۰ )
بمغلوبين عاجزين
ما تحرثون ( ٦۳ )
البَذر الذي تلقونه في الأرض
تزرعونه ( ٦٤ )
تنبتونه حتى يشتدّ و يبلغ الغاية
حُطاما ( ٦۵ )
هشيما متكسّرا لا يُنتفع به
تفكّهون ( ٦۵ )
تتعجّبون من سوء حاله و مصيره
إنّـا لمُغرمون ( ٦٦ )
مُهلكون بهلاك رزقنا
محرومون ( ٦۷ )
ممنوعون الرّزق بالكلّـيّـة
المُـزن ( ٦۹ )
السّحاب أو الأبيض منه
جعلناه أجاجا ( ۷۰ )
مِلحًا زُعاقا أو مُرّا لا يُمكن شربه
النّار التي تورون ( ۷۱ )
تقدحون الزّناد لاستخراجها
تذكرة ( ۷۳ )
تذكيرا لنار جهنّم
متاعا للمقوين ( ۷۳ )
مَنفعَة للمسافرين في القـَـوَاءِ (القـَـفـْـر) أو المُحتاجين إليها
فلا أقسم ( ۷۵ )
فأقسِم و "لا" مزيدة للتأكيد
بمواقع النجوم ( ۷۵ )
بمغاربها . أو منازلها
إنّه لقرآن كريم ( ۷۷ )
نـفّـاعٌ جمّ المنافع . أو رفيع القدْر
كتاب مكنون ( ۷۸ )
مستور مصون عند الله في اللوح المحفوظ من السّوء
لا يَمسّه إلا المطهّرون ( ۷۹ )
صفة أخرى للقرآن
أنتم مُدهنون ( ۸۱ )
مُتهاونون أو مكذبون
تجعلون رزقكم ( ۸۲ )
شكركم على الإنعام به
بلغت الحلقوم ( ۸۳ )
بلغت الرّوح الحلقوم عند المَوْت
نحن أقرب إليه ( ۸۵ )
بعِلمِنا و قدْرَتنا
غير مَدينين ( ۸٦ )
غير مربوبين مقهورين
فرَوْح ( ۸۹ )
فله استراحة أو رحمة
ريحان ( ۸۹ )
رزق حسن
فـنُـزُلٌ ( ۹۳ )
فله قِرًى و ضِيافة
حميم ( ۹۳ )
ماءٍ تناهت حرارته
تصلِية جحيم ( ۹٤ )
مُقاسات لحرّ النار أو إدخال فيها
