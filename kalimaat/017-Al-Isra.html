۱۷ - سورة الإسراء
( ۱ ) سبحان الذي :تنزيها لله و تعجيبا من قدرته
( ۱ ) أسرى بعبده :جعَلَ البراق يسري به صلى الله عليه و سلم
( ۱ ) لِنُريه . . :لِنرفعه إلى السّماء فنُريه . .
( ۲ ) وكيلا :ربّا تكلون إليه أموركم
( ۳ ) ذرّية . . :أخصُّ ذرّية أو يا ذرّية
( ٤ ) قضينا إلى بني إسرائيل :أوحينا إليهم و أعلمناهم بما سيقع منهم من الإفساد مرّتين
( ٤ ) لتعلُنّ :لَتفرطُنّ في الظلم و العدوان
( ۵ ) وعد أولاهما :العقاب الموعود على أولاهما
( ۵ ) أولي بأس :ذوي قوّة و بطش في الحروب
( ۵ ) فجاسوا :تردّدوا لطلبكم باستقصاء
( ۵ ) خلال الدّيار :وسطها
( ٦ ) الكَرّة :الدّولة و الغَلَبَة
( ٦ ) أكثر نفيرا :أكثر عددا أو عشيرة من أعدائكم
( ۷ ) ليسوءوا وُجوهكم :ليُحزنوكم حزنا يبدو في وجوهكم
( ۷ ) ليُتبّروا :ليُهلكوا و يُدمّروا
( ۷ ) ما علوْا :ما استولوا عليه
( ۸ ) حصيرا :سجنا أو مهادا و فراشا
( ۹ ) هي أقوم :أسدّ الطرق (ملة الإسلام – و التوحيد)
( ۱۲ ) الليل و النّهار :نفسهما أو نيّرى الليل و النّهار
( ۱۲ ) فمحوْنا آية الليل :خلقنا القمر مطموس النّور مظلما
( ۱۲ ) آية النّهار مُبصرة :الشمس مضيئة منيرة للأبصار
( ۱۳ ) ألزمناه طائره :عمله المقدّر عليه لا ينفكّ عنه
( ۱٤ ) حسيبا :حاسبا و عادّا . أو محاسبا
( ۱۵ ) لا تزر وازرة . . :لا تحمل نفس آثمةٌ . .
( ۱٦ ) أمَرنا مُترفيها :أمرنا متنعّميها بطاعة الله
( ۱٦ ) ففسقوا فيها :فتمرّدوا و عَصوْا
( ۱٦ ) فدمّرناها :استأصلناها و محونا آثارها
( ۱۷ ) القرون :الأمم المكذبة
( ۱۸ ) يصلاها :يدْخلها . أو يقاسي حرّها
( ۱۸ ) مدحورا :مطرودا مُبعدا من رحمة الله
( ۲۰ ) كُلاّ نمِدّ :نزيد من العطاء مرّة بعد أخرى
( ۲۰ ) محظورا :ممنوعا عمّن يريده تعالى
( ۲۲ ) مخذولا :غير منصور و لا مُعان من الله
( ۲۳ ) قضى ربّك :أمر و ألزم و حَكَم
( ۲۳ ) أفّ :كلمة تضجُّر و كراهيّة و تبرّم
( ۲۳ ) لا تنهرهما :لا تزجرهما عمّا لا يعجبك
( ۲۳ ) قولا كريما :حسنا جميلا ليّنا
( ۲۵ ) للأوّابين :للتّوابين مما يفرُط منهم
( ۲۹ ) يَدَك مغلولة :كناية عن الشّح
( ۲۹ ) تبسطها كلّ البسط :كناية عن التّبذير و الإسراف
( ۲۹ ) محسورا :نادما أو مُنقطعا بك مُعدما
( ۳۰ ) يقْدِر :يضيّقه على من يشاء لحكمة
( ۳۱ ) خشية إملاق :خوف فقر و فاقة
( ۳۱ ) خِطئاً كبيرا :إثما عظيما
( ۳۳ ) سُلطانا :تسلّطا على القاتل بالقصاص أو الدّيّة
( ۳٤ ) يبلغ أشدّه :قوّته على حفظ ماله و رشده فيه
( ۳۵ ) بالقسطاس المستقيم :بالميزان العدل
( ۳۵ ) أحسن تأويلا :مآلا و عاقبة
( ۳٦ ) لا تَقْــف :لا تتبع
( ۳۷ ) مرَحا :فرحا و بطرا و اختيالا و فخرا
( ۳۹ ) مدحورا :مُبعدا من رحمة الله
( ٤۰ ) أفصأفاكم ربّكم :أفضّلكم ربّكم فخصّكم ؟
( ٤۱ ) صرّفنا :كرّرنا القول بأساليب مختلفة
( ٤۱ ) نفورا :تباعُدا و إعراضا عن الحقّ
( ٤۲ ) لابتغوْا :لَطلبوا
( ٤۲ ) سبيلا :بالمُغالبة و الممانعة
( ٤۵ ) حجابا مستورا :ساترا أو مستورا عن الحسّ
( ٤٦ ) أكنّة :أغطية كثيرة مانعة
( ٤٦ ) وقرا :صمما و ثقلا في السّمع عظيما
( ٤۷ ) هم نجوى :مُتناجون في أمرك فيما بينهم
( ٤۷ ) مسحورا :مغلوبا على عقله بالسِّحر أو ساحرا
( ٤۹ ) رُفاتا :أجزاءً مفتتة . أو ترابا أو غُبارا
( ۵۱ ) يكبر :يعظم عن قبول الحياة كالسّماوات
( ۵۱ ) فطركم :أبدعكم و أحدثكم
( ۵۱ ) فسينغظون . . :يُحرّكون استهزاءً . .
( ۵۲ ) بحمده :مُنْقادين انْنقياد الحامدين له
( ۵۳ ) ينزغ بينهم :يُفسد و يُهيج الشّرّ بينهم
( ۵٤ ) وكيلا :موكلا إليك أمرُهم
( ۵۵ ) زبورا :كتابا فيه تحميد و تمجيد و مواعظ
( ۵٦ ) تحويلا :نقلَه إلى غيركم ممّن لم يعبُدهم
( ۵۷ ) الوسيلة :القربة بالطاعة و العبادة
( ۵۹ ) مُبصرة :آية بيّنةً واضحةً
( ۵۹ ) فظلموا بها :فكروا بها ظالمين فأُهلكوا
( ٦۰ ) أحاط بالنّاس :علما و قدْرة فهم في قبضته تعالى
( ٦۰ ) الشّجرة الملعونة :شجرة الزّقوم (جعلناها فتنة)
( ٦۰ ) طُغيانا :تجاوزا للحدّ في كفرهم و تمرّدا
( ٦۲ ) أرأيتَك :أخبرني
( ٦۲ ) لأحتنكنّ ذرّيّته :لأستولينّ عليهم . أو لأستأصلنّهم بالإغواء
( ٦٤ ) استفزز :استخِفّ و استعجلْ و أزعِجْ
( ٦٤ ) أجلب عليهم :صِحْ عليهم و سُقهم
( ٦٤ ) بخيلك و رجلك :بكلّ راكب و ماشٍ في معاصي الله
( ٦٤ ) غرورا :باطلا و خِداعا
( ٦۵ ) عليهم سلطان :تسلّط و قدرة على إغوائهم
( ٦٦ ) يُزجي :يُجري و يسيّر و يسوق برفق
( ٦۸ ) أن يخسف بكم :يُغوّر و يُغيّب بكم تحت الثرى
( ٦۸ ) حاصبا :ريحا شديدة ترميكم بالحصباء
( ٦۹ ) قاصفا :عاصفا شديدا مُهلكا
( ٦۹ ) تبيعا :نصيرا أو مُطالبا بالثأر منّا
( ۷۱ ) بإمامهم :بمن ائتموا به أو بكتابهم
( ۷۱ ) فتيلا :قدْر الخيط في شِقّ النواة من الجزاء
( ۷۳ ) ليَفتنونك :ليوقِعونك في الفتنة و ليصرفونك
( ۷۳ ) لتفتري علينا :لتختلق و تتقوّل علينا
( ۷٤ ) تركنُ إليهم :تميل إليهم
( ۷۵ ) ضعف الحياة :عذابا مُضاعفا في الحياة الدّنيا
( ۷٦ ) ليستفزّونك :لَيستخفّونك و يُـزعِجونك
( ۷۷ ) تحويلا :تغييرا و تبديلا
( ۷۸ ) لدُلوك الشّمس :بعد أو عند زوالها عن كبد السّماء
( ۷۸ ) غَسق الليل :ظُلمته أو شِدّتِها
( ۷۸ ) و قرآن الفجر :و أقِم صلاة الصّبح
( ۷۹ ) فتهجّدْ :التّهجّد : الصّلاة ليلا بعد الإستيقاظ
( ۷۹ ) نافلة لك :فريضة زائدة خاصّة بك
( ۷۹ ) مقاما محمودا :مقام الشفاعة العُظمى
( ۸۰ ) مُدخل صدق :إدْخالا مرضيّا جيّدا في أموري
( ۸۰ ) سلطانا نصيرا :قهرا و عِزّا ننصر به الإسلام
( ۸۱ ) زهق الباطل :زال و اضمحلّ الشّرك
( ۸۲ ) خسارا :هلاكا بسبب كُفرهم به
( ۸۳ ) نأى بجانبه :لوَى عِطفهُ تكبّرا و عِنادا
( ۸۳ ) كان يئوسا :شديد اليأس و القنوط من رحمتنا
( ۸٤ ) شاكلته :مذهبــِـهِ الذي يُشاكلُ حاله
( ۸٦ ) وكيلا :منْ تعهّد يإعادته إليك
( ۸۸ ) ظهيرا :مُعينا
( ۸۹ ) صرّفنا :ردّدنا بأساليب مختلفة
( ۸۹ ) كلّ مثل :معنى غريب حسن بديع
( ۸۹ ) فأبى :فلمْ يرضَ
( ۸۹ ) كُفورا :جُحُودا للحقّ
( ۹۰ ) ينبوعا :عينا لا ينْضَبُ ماؤها
( ۹۲ ) كِسفا :قِطعا
( ۹۲ ) قبيلا :مُقابلة و عِيانا . أو جماعة
( ۹۳ ) زُخرف :ذهَبٍ
( ۹۷ ) خبتْ :سكَنَ لَهبُها
( ۹۷ ) سعيرا :لهبا و توقّدا
( ۹۸ ) رُفاتا :أجزاء مُفتّتة . أو تُرابا أو غُبارا
( ۱۰۰ ) قتورا :مُبالغا في البُخل
( ۱۰۱ ) مسحورا :مغلوبا على عقلك بالسّحر أو ساحرا
( ۱۰۲ ) بَصائر :بيّنات تُبَصِّر من يشهدها بصدقي
( ۱۰۲ ) مثبورا :هالكا أو مصروفا عن الخير
( ۱۰۳ ) يستفزّهم :يَستخِفّهُمْ و يُزعجهم للخروج
( ۱۰٤ ) لفيفا :جميعا مُختلطين
( ۱۰٦ ) فرقناه :بيّــنّاه و فصّلناه أو أنزلناه مُفرّقا
( ۱۰٦ ) على مُكث :على تُؤَدَة و تأنّ
( ۱۱۰ ) لا تُخافت بها :لا تسرّ بها حتى لا تُسمع مَن خلفك
