۳۹ - سورة الزمر
( ۲ ) مُـخْـلصا :مُمَحّضًَا له الطاعة و العبادة
( ۳ ) زلفى :تقريبا
( ٤ ) سُبْحانه :تنزيها له عن اتخاذ الولد
( ۵ ) يُكوّر الليل على النّهار :يَلفّه على النهار لفّ اللباس على اللابس فيستره فتظهَر الظلمة
( ٦ ) أنزَلَ لكم :أنشأ و أحْدث لأجْلكم
( ٦ ) من الأنعام :الإبل و البقر و الضّأن و المعز
( ٦ ) ظلمَات ثلاثٍ :ظلمة البطن و الرّحم و المشيمة
( ٦ ) فأنّى تـُـصْـرَفون ؟ :فكيف تصرفون عن عبادته ؟
( ۷ ) لا تزر وازرَةٌ . . :لا تحْمل نـفسٌ آثمة . .
( ۸ ) مُـنِـيـبا إليه :راجعا إليه ، مستغيثا به
( ۸ ) خَوّله نِعْمَة :أعطاه نعمة عظيمة تفضلا و إحسانا
( ۸ ) أندَادًا :أمثالا يَعبدها من دونه تعالى
( ۹ ) هوَ قـَـانتٌ :مطيع خاضع عابدٌ لله تعالى
( ۹ ) آناءَ الليل :سَاعاته
( ۱۰ ) بغَير حِساب :بلا نهاية لما يعطي أو بتوْسعة
( ۱٦ ) ظلـَـلٌ مِنَ النار :أطباق منها أ كثيرة متراكمة
( ۱۷ ) اجْـتـنـَبوا الطّـاغوتَ :الأوثان و المعبودات الباطلة
( ۱۷ ) أنـَابوا إلى الله :رجعوا إلى عبادته وحده
( ۱۹ ) حقّ عَليْه :وَجَبَ و ثبَت عليه
( ۲۰ ) لهمْ غرفٌ :منازل رفيعة عالية في الجنة
( ۲۱ ) فـَـسَـلكَه ينابيعَ :أدخله في عيون و مَجَار
( ۲۱ ) يَهيج :يَيبَس في أقصى غايته
( ۲۱ ) يَـجْـعَـلهُ حُطامًـا :يصيّره فتاتا هشيما متكسّرا
( ۲۲ ) فويْـلٌ :هلاك أو حسرة أو شدّة عذاب
( ۲۳ ) أحسَنَ الحديث :أبلغَه و أصْدقه و أوْفاه (القرآن)
( ۲۳ ) كِـتابًا مُـتـَشابها :في إعجازه و هدايته و خصائصه
( ۲۳ ) مَـثـانيَ :مكرّرا فيه الأحكام و المواعظ و القـَـصَص و غيرها
( ۲۳ ) تـَـقـْـشعرّ منه . . :تضْطرب و ترتعِدْ مِنْ قـَوَارعه . .
( ۲۳ ) تـَلين جلودهم :تسكن و تطمئنّ ليّـنة غير مُـنقـَبضة
( ۲٦ ) الخِـزيَ :الذلّ و الهَوَان
( ۲۸ ) عِـوَج :اختلاف و اختلال و اضطراب
( ۲۹ ) شرَكاءُ مُـتـَـشاكِسون :متنازعون شرسوا الطّباع
( ۲۹ ) سَلـَـمًا لرَجل :خالصا له مِنَ الشـّرِكَة و المُنازعة
( ۳۲ ) مَـثـوًى للكافرين :مأوًى و مُقامٌ لهمْ
( ۳۸ ) أفرأيتمْ :أخبروني
( ۳۸ ) حَسْبيَ الله :كافيّ في جميع أموري
( ۳۹ ) مَكانـَـتِـكمْ :حالتكم المُـتمَـكّـنين منها
( ٤۰ ) يُخْـزيه :يُذلـّه و يهينه
( ٤۰ ) يحلّ عليه :يَجبُ عليه
( ٤۲ ) يَتـَـوَفّى الأنفس :يقبضها عن الأبدان
( ٤٤ ) لله الشـّـفاعة جميعًا :لا يشفع أحدٌ عنده إلا بإذنه
( ٤۵ ) اشمأزّتْ :نـَـفَرتْ و انقبَضتْ عن التوحيد
( ٤٦ ) فَاطرَ . . :يا مُبْدع ومخترع . .
( ٤۷ ) يَحتـَسِبونَ :يظنّونه و يتوقـّعونه
( ٤۸ ) حَق بهمْ :نزل أو أحَاط بهم
( ٤۹ ) خوّلـنـَاه نِـعْمَة :أعطيناه إيّـاه تـَفضّلا و إحساناً
( ٤۹ ) هَيَ فتـْنة :تِلك النّعمة امتحان و ابتلاء
( ۵۱ ) بمعجزين :بفائتين من العَذاب بالهَرَب
( ۵۲ ) يَقدر :يُضيّـقه على من يشاء بحكمته
( ۵۳ ) أسْرَفوا :تجاوَزوا الحدّ في المعاصي
( ۵۳ ) لا تـَـقـْـنـَـطوا :لا تيْـأسوا
( ۵۳ ) الذنوب جَميعا :إلاّ الشّرك
( ۵٤ ) أنيبوا إلى ربّـكم :ارجعوا إليه بالتـّوبة و الطاعة
( ۵٤ ) أسْـلموا له :أخلصوا له عِبَادَتكم
( ۵۵ ) بَغتة :فجْـأة
( ۵٦ ) يا حَسرَتا :يا نـَـدَمِي و يا حُـزْني
( ۵٦ ) فرّطت :قصّرْت
( ۵٦ ) في جنب الله :في طاعته و أمره و حَقّه تعالى
( ۵٦ ) السّاخرين :المستهزئين بدينه و كتابه و أهله
( ۵۸ ) كَرّة :رَجْعة إلى الدنيا
( ٦۰ ) مَثوًى للمتكبّرين :مَأوًى و مُقامٌ لهم
( ٦۱ ) بمفازتهمْ :بفوْزهم و ظفرهِمْ بالبغيَة
( ٦۳ ) له مقاليد. . :مَفاتيح أو خزائن . .
( ٦۵ ) لـيَـحبطنّ عملك :ليبْطلنّ عَمَلك و يَفسدنّ
( ٦۷ ) ما قدَروا الله . . :ما عَرَفوه . أو ما عظّـموه . .
( ٦۷ ) قبْضته :مِلكه و في مَقدوره و تصرّفه
( ٦۷ ) مَطْويات بيمينه :بقدرته كطيّ السّجل للكتب
( ٦۸ ) الصّور :القرن الذي يَنفخ فيه إسرافيل
( ٦۸ ) فصَعِـق :ماتَ. و هي النّـفخة الأولى
( ٦۹ ) وُضِع الكتاب :أعطيَتْ صحف الأعمال لأربابها
( ۷۱ ) زمرًا :جَمَاعات متفرقة متتابعة
( ۷۱ ) حقـّـتْ :وَجَبَتْ و ثـبَـتـتْ
( ۷۳ ) طِبْـتمْ :طَهرتم مِن دنس المعاصي
( ۷٤ ) صَدَقـَنـَا وَعده :أنجزنا ما وعَدَنا من النّعيم
( ۷٤ ) نتبوأ :ننزل
( ۷۵ ) حَافّـين :مُحْدقينَ مُحيطين
