۲۰ - سورة طه
( ۲ ) لِتشقى :لتتعب بالإفراط في مُكابدة الشدائد و التأسّف على قومك
( ۵ ) على العرش استوى :استواءً يليق به تعالى
( ٦ ) ما تحت الثـّرى :ما وراه التراب. أو ما وراء الأرض
( ۷ ) أخفى :حديث النّـفـس و خواطرها
( ۱۰ ) آنست نارا :أبصرتها بوُضوح
( ۱۰ ) بقبس :بشعلة نار مقبوسة على رأس عود
( ۱۰ ) هدًى :هاديا يهديني إلى الطريق
( ۱۲ ) المقدّس :المُطهّر أو المبارك
( ۱۲ ) طُوى :اسم للوادي
( ۱۵ ) أكاد أخفيها :أقربُ أن أسترها من نفسي
( ۱٦ ) فتـَردى :فتهلِك
( ۱۸ ) أتوكّـأ عليها :أتحامل عليها في المشي و نحوه
( ۱۸ ) أهشّ بها :أخبطُ بها الشجر ليتساقط الورق
( ۱۸ ) مآرب أخرى :حاجات و منافع أخرى
( ۲۰ ) حيّة تسعى :تمشي بسرعة و خفة
( ۲۱ ) سيرتها الأولى :إلى حالتها التي كانت عليها
( ۲۲ ) إلى جناحك :إلى جنبك تحت العضد الأيسر
( ۲۲ ) بيضاء :لها شعاع الشّمس
( ۲۲ ) غير سُوء :غير داءٍ بَرَص و نحوه
( ۲٤ ) طغى :جاوز الحدّ في العتوّ و التجبّر
( ۲۹ ) وزيرا :ظهيرا و مُعينا
( ۳۱ ) أزْري :ظهري أو قوتي
( ۳٦ ) أوتيت سؤْلك :أعطِيت مَسْـئولك و مطلوبك
( ۳۹ ) فاقذفيه في اليمّ :فألقيه و اطرحيه في نهر النّيل
( ۳۹ ) لتـُصنع على عيني :لترَبّي بمُراقبتي أو بمرأى منّي
( ٤۰ ) من يكفله :من يضمّه إليه و يحفظه و يُربّيه
( ٤۰ ) تقرّ عينها :تـُـسَرّ بلقائك
( ٤۰ ) فتنّاك فتونا :خلّصناك من المِحن تخليصا
( ٤۰ ) جئت على قدر :على وَفق الوقت المقدّر لإرسالك
( ٤۱ ) اصطنعتك لنفسي :اصطفيتك لرسالتي و إقامة حُجّتي
( ٤۲ ) لا تنيا في ذكري :لا تفترا في تبليغ رسالتي
( ٤۵ ) يفرُط علينا :يعجل علينا بالعقوبة
( ٤۵ ) يطغى :يزداد طُغيانا و عُـتوّاو جراءة
( ٤٦ ) إنّـني معكما :حافظُكما و ناصركما
( ۵۰ ) خَلـْقـَه :صورَته اللائقة بخاصّــته و منفعته
( ۵۰ ) هَدَى :أرشدَه إلى ما يصلح له
( ۵۱ ) فما بال القرون ؟ :فما حال و ما شأن الأمم؟
( ۵۲ ) لا يضلّ ربي :لا يغيب عن علمه شيء ما
( ۵۳ ) مهدا :كالفراش الذي يُوطّأ للصّبي
( ۵۳ ) سُبُـلا :طُرُقا تسلكونها لقضاء مآربكم
( ۵۳ ) أزواجا :أصنافا أو ضروبا
( ۵۳ ) شتـّى :مُختلفة الصّـفات و الخصائص
( ۵٤ ) لألي النـّـُـهى :لأصحاب العقول و البصائر
( ۵٦ ) أبى :امتنع عن الإيمان و الطّاعة
( ۵۸ ) مكانا سُوى :وَسَطا أو مُستويا من الأرض
( ۵۹ ) يوم الزّينة :يوم عيدكم (يوم مشهود)
( ٦۰ ) فجمع كيده :سحَرته الذين يكيد بهم
( ٦۱ ) ويلكم :دُعاء عليهم بالهلاك
( ٦۱ ) فيُسحتكم :فيستأصلكم و يُبيدكم
( ٦۲ ) أسرّوا النّجوى :أخفوا التّناجي أشدّ الإخفاء
( ٦۳ ) بطريقتكم المُثلى :بسنّتكم و شريعتكم الفضلى
( ٦٤ ) فأجمعوا كيدَكم :فأحكموا سِحركم و اعزموا عليه
( ٦٤ ) أفلح :فاز بالمطلوب
( ٦۷ ) فأوجس في نفسه :أضمَرَ. أو وَجَد و أحسّ في نفسه
( ٦۹ ) تلقف :تبتلعْ و تلتقـِمْ بسرعة
( ۷۲ ) و الذي فطرنا :أبدعنا و أوجدنا و هو الله تعالى
( ۷٦ ) تزكّى :تطهّر من دنس الشّرك و الكفر
( ۷۷ ) أسر بعبادي :سِرْ ليلا بهم من مصر
( ۷۷ ) يَــبَـسا :يابسا لا ماء فيه و لا طين
( ۷۷ ) لا تخاف دَرَكا :لا تخشى إدراكا و لحاقا أو تبعة
( ۷۷ ) لا تخشى :الغرق من الأمام
( ۷۸ ) فغشِــيهم :علاهم و غمرهم
( ۸۰ ) المنّ :مادّة صمغية حلوة كالعسل
( ۸۰ ) السّلوى :الطائر المعروف بالسّماني
( ۸۱ ) لا تطغوْا :لا تكفروا نعمه أو لا تظلموا
( ۸۱ ) فيحلّ عليكم :فيجب عليكم و يلزمكم
( ۸۱ ) هَوَى :هلك . أو وَقع في الهاوية
( ۸۳ ) ما أعجلك؟ :ما حملك على العجلة ؟
( ۸۵ ) فتـنـّـا قومك :ابتليناهم . أو أوقعناهم في فتنة
( ۸٦ ) أسِفا :حزينا . أو شديد الغضب
( ۸٦ ) موعدي :وعدكم لي بالثبات على ديني
( ۸۷ ) بملكِنا :بقدرتِنا و طاقتِنا
( ۸۷ ) أوْزارا :أثقالا أو آثاما و تبـِعات
( ۸۷ ) من زينة القوم :من حليّ قبط مصر
( ۸۸ ) عِجلا جسدا :مُجسّدا : أي أحمر من ذهب
( ۸۸ ) لهُ خوار :صوت كصوت البقر
( ۹۲ ) ما منعَك :ما حمَلك و اضطرّك
( ۹۵ ) فما خطبُك ؟ :فما شأنك الخطير ؟
( ۹٦ ) بَصُرتُ :علِمتُ بالبصيرة
( ۹٦ ) أثر الرّسول :أثر فرس جبريل عليه السلام
( ۹٦ ) فنبذتها :ألقيتُها في الحليّ المُذاب
( ۹٦ ) سوّلت :زيّـنت و حسّـنت
( ۹۷ ) لا مِساس :لا تمسّـني و لا أمسّـك
( ۹۷ ) لنـنـْسِفـنّـه :لَنُذرّينّه
( ۱۰۰ ) وزرا :عقوبة ثقيلة على إعراضه
( ۱۰۲ ) زرقا :زُرق العيون . أو عُميا . أو عطاشا
( ۱۰۳ ) يتخافتون :يتسارّون و يتهامسون
( ۱۰٤ ) أنثلهم طريقة :أعدلهم و أفضلهم رأيا و مذهبا
( ۱۰۵ ) ينسفها :يقتلعها أو يفتتها و يقرّقها بالرّياح
( ۱۰٦ ) قاعا :أرضا ملساء لا نبات و لا بناء فيها
( ۱۰٦ ) صفصفا :أرضا مُستوية أو لا نبات فيها
( ۱۰۷ ) عِوَجا :مكانا مُنخفضا. أو انخفاضا
( ۱۰۷ ) أمْـتـا :مكانا مُرتفعا . أو ارتفعا
( ۱۰۸ ) لا عِوَج له :لا يعوجّ له مدعُوٌ و لا يزيغ عنه
( ۱۰۸ ) هَمْسا :صوتا خفيّا خافتا
( ۱۱۱ ) عنـَتِ الوجوه :ذلّ النّاس و خضعوا
( ۱۱۱ ) للحيّ :الدّائم الحياة بلا زوال
( ۱۱۱ ) القيّوم :الدّائم القيام بتدبير الخلق
( ۱۱۱ ) حمل ظلما :شركا و كفرا
( ۱۱۲ ) هضنا :نقصا من ثوابه
( ۱۱۳ ) صرّفنا فيه :كرّرنا فيه بأساليب شتى
( ۱۱۳ ) ذِكرا :عِظة و اعتبارا
( ۱۱٤ ) أن يُقضى إليك :أن يُفرغ و يُتمّ إليك
( ۱۱۵ ) عَهـِدنا إلى آدم :أمرناه أو أوحينا إليه
( ۱۱٦ ) أبى :امتنع من السّجود استكبارا
( ۱۱۸ ) لا تعرَى :لا يُصيبك عُريٌ عن الملابس
( ۱۱۹ ) لا تضحى :لا تبرز للشّمس فيصيبك حرّها
( ۱۲۰ ) لا يبلى :لا يزول و لا يفنى
( ۱۲۱ ) سوآتهما :عوراتهما
( ۱۲۱ ) طفقا يخصفان :أخذا يُلصقان و يلزقان
( ۱۲۱ ) عصى آدم :خالفا النّهي سهوا أو بتأوّل
( ۱۲۱ ) فغوى :فضّل عن مطلوبه أو عن النّهي
( ۱۲۲ ) اجتباه :اصطفاه للنّبوّة و قرّبه
( ۱۲٤ ) معيشة ضنكا :ضيّقة شديدة (في قبره)
( ۱۲۸ ) أفلم يهد لهم :أغفلوا فلم يُبيّن لهم مآلهم
( ۱۲۸ ) كم أهلكنا :كثرة إهلاكنا الأمم الماضية
( ۱۲۸ ) لأولى النـّـهى :لذوي العقول و البصائر
( ۱۲۹ ) لكان لزاما :لكان إهلاكهم عاجلا لازما
( ۱۲۹ ) أجل مسمّى :يوم القيامة (عطف على كلمة)
( ۱۳۰ ) سبّح بحمد ربّك :صلّ و أنت حامد لربّك
( ۱۳۰ ) آناء الليل :ساعاته
( ۱۳۱ ) أزواجا منهم :أصنافا من الكفّار
( ۱۳۱ ) زهرة الحياة الدنيا :زينتها و بهجتها
( ۱۳۱ ) لِنفتنهم فيه :لنجعله فتنة لهم و ابتلاء
( ۱۳۳ ) بيّــنة :هي القرآن المُعجـِز أم الآيات
( ۱۳٤ ) من قبله :من قبل الإثبات بالبيّنة
( ۱۳٤ ) نخزى :نَفتـَضح في الآخرة بالعذاب
( ۱۳۵ ) مُتربّصٌ :مُنتظر مآله
( ۱۳۵ ) الصّراط السّويّ :الطّريق المستقيم
