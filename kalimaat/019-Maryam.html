۱۹ - سورة مريم
( ۳ ) نداءً خفيا :دعاءً مستورا لم يسمعه أحد
( ٤ ) وهَنَ العظم :ضَعُـفَ و رَقّ
( ٤ ) شقيّا :خائبا في وقت مّا
( ۵ ) خفت الموالي :أقاربي العَصبَة و كانوا شرار اليهود
( ۵ ) وليّا :ابناً يلي الأمر بعدي
( ٦ ) رضيّا :مرضيّا عندك قولا و فعلا
( ۸ ) أنّى يكون ؟ :كيف أو من أين يكون ؟
( ۸ ) عتيّا :حالةً لا سبيل إلى مُداواتها
( ۱۰ ) آية :علامة على تحقّق المسئول لأشكرك
( ۱۰ ) سويّا :سليما لا خرس بك و لا علة
( ۱۱ ) من المحراب :المُصلّى أو الغرفة التـّي يتعبّد فيها
( ۱۱ ) بُكرة و عشيّا :طرفي النّهار
( ۱۲ ) الحكم :فهم التّوراة و العبادة
( ۱۳ ) حنانا :رحمة و عطفا على النّاس
( ۱۳ ) زكاة :بركة . أو طهارة من الذنوب
( ۱۳ ) كان تقيّا :مُطيعا مجتنبا للمعاصي
( ۱٤ ) برّا بوالديه :كثيرَ البرّ و الإحسان إليهما
( ۱٤ ) جبّارا عصيّا :مُتكبّرا مخالفا أمر ربّه
( ۱٦ ) انتبذت :اعتزلت و انفردتْ
( ۱۷ ) حجابا :سِترا
( ۱۷ ) روحنا :جبريل عليه السّلام
( ۱۷ ) بشرا سويّا :إنسانا مُستوي الخلق تامّه
( ۱۹ ) غُلاما زكيّا :مُزكّى مُطهّرا بالخِلقة
( ۲۰ ) يغيّا :فاجرة تبغي الرّجال
( ۲۲ ) مكانا قصيّا :بعيدا من أهلها وراء الجبل
( ۲۳ ) فأجاءها المخاض :فألجأها و اضطرّها وجع الولادة
( ۲۳ ) نسيا منسيّا :شيئا حقيرا متروكا لا يخطر بالبال
( ۲٤ ) فناداها :جبريل أو عيسى عليهما السلام
( ۲٤ ) سريّا :جدولا أو غلاما سامِيَ القـَدْر
( ۲۵ ) رُطبًا جنيّا :صالحا للاجْتِناء . أو طريّا
( ۲٦ ) قرّي عينا :طيبي نفسا و لا تحزني
( ۲۷ ) شيئا فريّا :عظيما مُنكرا
( ۲۹ ) كان في المهد صبيّا :وُجد في فِراش الصّبية رضيعا
( ۳۲ ) برّا بوالدتي :بارّا بها مُحسنا مُكرما
( ۳٤ ) قول الحقّ :كلمة الله لخلقه بقوله كنْ
( ۳٤ ) يمترون :يشكّون أو يتجادلون بالباطل
( ۳۵ ) قضى أمرا :أراد أن يُحدِثه
( ۳۸ ) أسمع بهم و أبصر :ما أسمعهم و ما أبصرهم
( ۳۹ ) يوم الحسرة :النّدامة الشّديدة على ما فات
( ٤۳ ) صراطا سويّا :طريقا مُستقيما مُنجيا من الضلال
( ٤٤ ) عصيّا :كثير العصيان
( ٤۵ ) وليّا :قرينا تليه و يليك في النّار
( ٤٦ ) اهجرني مليّا :اجتنبني و فارقني دهرا طويلا
( ٤۷ ) حفيّا :برّا لطيفا أو رحيما مُكرما
( ٤۸ ) شقيّا :خائبا ضائع السّعي
( ۵۰ ) لسان صِدق :ثناءً حسنا في أهل كلّ دين
( ۵۱ ) كان مخلصا :أخلصه الله و اصطفاه
( ۵۲ ) قرّبناه نجيّا :مُناجيا لنا
( ۵۸ ) اجتبينا :اصطفيْنا و اخترنا للنّبوّة
( ۵۸ ) بُكيّا :باكين من خشية الله
( ۵۹ ) خلفٌ :عَـقِـبٌ سَوْءٍ
( ۵۹ ) يلقون غيّا :جزاء الغيّ . أو واديا في جهنّم
( ٦۱ ) مأتيّا :آتيا أو مُــنـَـجَّــزًا
( ٦۲ ) لغوا :قبيحا أو فُضولا من الكلام
( ٦۵ ) سميّا :مُضاهيا في ذاته و صفاته : لا
( ٦۸ ) جثيّا :باركِين على رُكبهم لشدّة الهول
( ٦۹ ) عتيّا :عِـصيانا ، أو جراءة أو فُجورا
( ۷۰ ) صليّا :دُخولا أو مُقاساة لحرّها
( ۷۱ ) واردها :بالمرور على الصّراط الممدود عليها
( ۷۳ ) خيرٌ مُقاما :منزلا و سكنا
( ۷۳ ) أحسن نديّا :مجلسا و مُجتمعا
( ۷٤ ) قرْنٍ :أمّة
( ۷٤ ) أحسن أثاثا :متاعا من الفَرْش و الثياب و غيرها
( ۷٤ ) رئيا :منظرا و هيئة
( ۷۵ ) فليمدد له :يُمهله استدراجا
( ۷۵ ) أضعف جندا :أقلّ أعوانا و أنصارا
( ۷٦ ) خير مردّا :مرجعا و عاقبة
( ۷۷ ) أفرأيت :أخبرني
( ۷۷ ) أطّـلعَ الغيب :أعلم الغيب (استفهام)
( ۷۹ ) نمدّ له :نُطوّل له أو نزيدُهُ
( ۸۱ ) عزّا :شُفعاء و أنصارا يتعزّزون بهم
( ۸۲ ) ضدّا :ذلاّ و هوانا لا عزّا أو أعوانا عليهم
( ۸۳ ) تؤزّهم أزّا :تُغريهم بالمعاصي إغراء
( ۸۵ ) وفدا :ركبانا . أو وافدين استرفاداً
( ۸٦ ) وردا :عِطاشا . أو كالدّوابّ التي تَردُ الماء
( ۸۹ ) شيئا إدّا :منكرا فظيعا
( ۹۰ ) يتفطّرن منه :يتشـقـّـقـْن و يتفــتـّــتـْن من شناعته
( ۹۰ ) تخرّ الجبال هدّا :تسقط مهدودة عليهم
( ۹٦ ) وُدّا :مودّة و محبّة في القلوب
( ۹۷ ) قوما لدّا :شديدي الخصومة بالباطل
( ۹۸ ) قرْن :أمّة
( ۹۸ ) تحِسّ :تجد . أو ترى . أو تعلم
( ۹۸ ) ركزا :صوْتا خفيّا
