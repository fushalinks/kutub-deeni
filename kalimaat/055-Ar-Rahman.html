۵۵ - سورة الرحمن
علـّـمَ القرآن ( ۲ )
عَـلـّـم الإنسان القرآن
بحُـسبان ( ۵ )
يجريان بحساب مقدّر في بروجهما
النّجم ( ٦ )
النبات الذي ينجم و لا ساق له
يسجدان ( ٦ )
ينقادان لله فيما خُـلِقا له
وضع الميزان ( ۷ )
شرَع العدل و أمر به الخلق
ألا تطْغَوْا ( ۸ )
لئلاّ تتجاوزوا العدل و الحقّ
بالقسْط ( ۹ )
بالعدْل
لا تـُـخْـسِروا الميزان ( ۹ )
لا تنقصوا مَوْزون الميزان
الأرض وضعها ( ۱۰ )
خلقها محفوظة عن السماء
ذات الأكمام ( ۱۱ )
أوعية الثمر و هي الطلْع
ذو العَصْف ( ۱۲ )
القِـشْـر أو التـّـبْن أو الوَرَق اليابس
الرّيحان ( ۱۲ )
النبات المشموم الطيّب الرّائحة
آلاء ربّكما ( ۱۳ )
نعمِه تعالى
تكذبان ( ۱۳ )
تكفران أيها الثقلان
صلصال ( ۱٤ )
طين يابس يُسمع له صلصلة
كالفخّار ( ۱٤ )
هو الطين يُحرَق حتى يتحجّر
مارج ( ۱۵ )
لهبٍ صافٍ لا دخان فيه
مَرَج البحرين ( ۱۹ )
أرْسَل العـذب و المِلح في مجاريهما
يلتقِـيان ( ۱۹ )
يتجاوران أو يلتقي طرفاهما
بيْنهما برْزخ ( ۲۰ )
حاجز أرضي أو من قدرته تعالى
لا يبْغِيان ( ۲۰ )
لا يطغى أحدهما على الآخر بالمُـمَـازحَة
له الجوار ( ۲٤ )
السّـفـن الجارية
المنشئات ( ۲٤ )
المرفوعات الشّرُع (القلوع)
كالأعلام ( ۲٤ )
كالجبالالشّاهقة أو القصور
فان ( ۲٦ )
هالك
ذو الجلال ( ۲۷ )
العظمة و الاستغناء المطلق
الإكرام ( ۲۷ )
الفَـضْـل التام
في شأن ( ۲۹ )
يأتي بأحوالٍ و يذهب بأحوالٍ بالحكمة
سنفرُغ لكم ( ۳۱ )
سنقصِد لمحاسَبتكم بعد الإهمال
أيها الثقلان ( ۳۱ )
الإنس و الجنّ
تنفذوا ( ۳۳ )
تخرجوا هربا من قضائي
فانفذوا ( ۳۳ )
فاخرجوا (أمر تعجيز)
بسلطان ( ۳۳ )
بقوّة و قهر، و هيهات . . !
شواظ ( ۳۵ )
لهب خالص لا دخان فيه
نحاس ( ۳۵ )
صُـفْـرٌ مُذاب أو دخان بلا لهَب
فكانت وردة ( ۳۷ )
كالوردة في الحمرة
كالدّهان ( ۳۷ )
كدُهْن الزيت في الذوبان
بسيماهم ( ٤۱ )
بسواد الوجوه ، و زرقة العيون
فيؤخذ بالنّواصي ( ٤۱ )
بشعور مقدّم الرّءوس
حميم آن ( ٤٤ )
ماءٍ حارتناهى حرّه
جنّـتان ( ٤٦ )
بستان داخل القصر و آخر خارجه
ذواتا أفنان ( ٤۸ )
أغصان . أو أنواع من الثمار
عينان ( ۵۰ )
التسْـنيم و السلسبيل
زوجان ( ۵۲ )
صِنفان : معروف و غريب
استبرق ( ۵٤ )
غليظ الديباج
جنى الجنـّـتـيْـن ( ۵٤ )
ما يُجنى من ثمارهما
دان ( ۵٤ )
قريب من يد المتناول
قاصرات الطّرف ( ۵٦ )
قصَرْنَ أبصارَهنّ على أزواجهنّ
لم يطمثهنّ ( ۵٦ )
لمْ يفـتضّهنّ قبل أزواجهنّ
و من دونهما جنـّـتان ( ٦۲ )
أعلى أو أدنى من السابقـتـيْن
مدهامّـتان ( ٦٤ )
خضراوان شديدتا الخضرة
نضّاختان ( ٦٦ )
فوّارتان بالماء لا تنقطعان
خيرات حسان ( ۷۰ )
خيّرات الأخلاق حِسان الوُجوه
حور ( ۷۲ )
نساءٌ بيض حِسان
مقصورات في الخِيام ( ۷۲ )
مخدّرات في بيوت من اللؤلؤ
رفرف ( ۷٦ )
وسائد أو فُرُش مُرتفعة
عبقريّ ( ۷٦ )
بُسطٍ ذات خمْل رقيق
تبارك ( ۷۸ )
تعالى . أو كثـُـرَ خيْرُه و إحسانه
ذي الجلال ( ۷۸ )
العظمة و الإستغناء المُطلق
