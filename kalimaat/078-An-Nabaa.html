۷۸ - سورة النبإ
عمّ ؟ ( ۱ )
عن أيّ شيء عظيمِ الشّـأن؟
عن النّبإ العظيم ( ۲ )
عن القرآن أو البَعْث
كلاّ ( ٤ )
رَدْعٌ و زَجْر عن الإختلاف فيه
الأرض مِهادًا ( ٦ )
فِـرَاشًـا مُوَطّـأ للاستقرار عليها
الجبال أوتادًا ( ۷ )
كالأوتاد للأرض لئلاّ تميد
خلقناكم أزواجًا ( ۸ )
أصنافا ذكورًا و إناثـا للتـّـناسُـل
نومَكم سُباتا ( ۹ )
قـَـطْعًا لأعمالِكُمْ و راحة لأبْدانِكم
الليل لِباسًا ( ۱۰ )
ساتِرا لكم بظلمتِه كاللـّـباس
النّـهار معاشا ( ۱۱ )
تحَصّلون فيه ما تعيشون به
سَبْـعًا شٍـدادًا ( ۱۲ )
سمواتٍ قويّـاتٍ مُحْكَمات
سِرَاجا وهّاجا ( ۱۳ )
مِصْباحا مُنيرا وقـّـادًا (الشّمس)
المُعصِرَات ( ۱٤ )
السّحائب التي حان لها أن تـُمْطِر
ماءً ثجّاجا ( ۱٤ )
مُنـْـصَـبّـا بكثرةٍ مع التـّـتابُع
جنّاتٍ ألفافا ( ۱٦ )
بساتين مُـلتـَـفّة الأشجار
فتأتون أفواجا ( ۱۸ )
أمَمًـا أو جماعات مُختلفة الأحوال
فكانت أبوابا ( ۱۹ )
صارَتْ ذات أبواب و طرُق
فكانت سرابا ( ۲۰ )
فالسّراب الذي لا حَقيقة له
كانت مِرْصادًا ( ۲۱ )
مَوْضِعَ ترَصّدٍ و ترقّب للكافرين
للطاغين مآبًـا ( ۲۲ )
مَرْجـِعًـا و مأوى لهم
أحقابـًا ( ۲۳ )
دهورًا ممتابعة لا نهاية لها
بَرْدًا ( ۲٤ )
نـَوْمًا أو رَوْحًا من حرّ النّـار
حميما ( ۲۵ )
ماءً بالغا نِهاية الحرارة
غسّـاقا ( ۲۵ )
صديدا يسيل من جُـلودِهم
جَزَاءً وفاقـًا ( ۲٦ )
جزَيْنـاهم جَزَاءً مُوافِـقا لأعمالِهِم
كِذابا ( ۲۸ )
تكذيبًـا شديدا
أحصيْناه كِتابا ( ۲۹ )
حَفِظناه و ضبطناه مكتوبًـا
مَفازًا ( ۳۱ )
فوْزًا و ظفَـرًا بكلّ محْبوب
كواعِب ( ۳۳ )
فتيات ناهِدات (نساء الجَنّة)
أترابًا ( ۳۳ )
مُسْتـَـوياتٍ في السّن
كأسًـا دِهاقـا ( ۳٤ )
مُتـْـرَعَة مليئة من خَمْر الجَنّة
لغْوًا ( ۳۵ )
كلاما غير مُعْـتدّ به . أو قبيحًـا
كِذابا ( ۳۵ )
تكذيبًا
عَطاءً حِسابا ( ۳٦ )
إحسانا كافِـيا أو كثيرا
خِطابـًا ( ۳۷ )
إلا بإذنه
الرّوح ( ۳۸ )
جبريل عليه السلام
مآبـا ( ۳۹ )
مَرْجـِـعًـا بالإيمان و الطاعة
كُـنْت تـُـرَابًـا ( ٤۰ )
