٦٦ - سورة التحريم
ما أحلّ الله لك ( ۱ )
شرب العسل
تبْـتغي ( ۱ )
تـَـطلب
تحلّة أيْمانكمْ ( ۲ )
تحليلها بالكفّارة
الله مولاكمْ ( ۲ )
ناصِركم و متولّي أموركم
نبّـأت به ( ۳ )
أخْـبَرَتْ به غيرها
أظهَرَه الله عليه ( ۳ )
أطلعه الله تعالى على إفشائه
صَغَتْ قلوبكما ( ٤ )
مَالتْ عن حَقـّه صلى الله عليه و سلم عليكما
تظاهَرَا عليه ( ٤ )
تتعاونـَا عليه بما يسوءُه
هو مولاه ( ٤ )
وَليّـه و ناصِره
ظهير ( ٤ )
فوْجٌ مُظاهر معينٌ له
قانتات ( ۵ )
مطيعات خاضعاتٍ لله
سائحات ( ۵ )
مهاجرَاتٍ . أو صائمات
قوا أنفسكم ( ٦ )
جَـنّـبوها بالطاعات
غِلاظ شِداد ( ٦ )
قساةٌ أقوياء و هم الزبانية
توبة نـَـصوحا ( ۸ )
خالصة . أو صادقة . أو مقبولة
لا يُخزي الله النبي ( ۸ )
لا يُـذلّه بل يُعزّه و يكرمه
اغلظ عليهم ( ۹ )
شدِّدْ. أو اقـْـسُ عليهم
فخانتاهما ( ۱۰ )
بالنّـفاق أو بالنميمة
فلم يغنيا عنهما ( ۱۰ )
فـلمْ يَدْفَعا و لم يمْنعاعنهما
أحصنت فرجها ( ۱۲ )
عَـفّـتْ و صانته من الرّجال
من روحنا ( ۱۲ )
روحًا مِنْ خَلقِـنـَـا بلا توسّط أبٍ (عيسى عليه السلام)
من القانتين ( ۱۲ )
من القوم المطيعين لربّهم
