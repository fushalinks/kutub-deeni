۵٤ - سورة القمر
( ۱ ) انشقّ القمَر :قدِ انفلق فَـلـْـقـتيْن مُعْجزة له صلى الله عليه و سلم
( ۲ ) سِـحْـرٌ :دائمٌ . أوْ مُـحْـكمٌ أو ذاهبٌ
( ۳ ) مسْـتـَقِـرّ :مُـنـْــتـهٍ إلى غاية يستقـرّ عليها
( ٤ ) مُزدجَرٌ :ازدجار و انتِهارٌ و ردْعٌ عمّا هم فيه من الكفر و الضلال
( ۵ ) النّذر :الرّسل أو الأمور المُـخَـوِّفـَـة لهم
( ٦ ) شيءٍ نُـكُـر :منكر فظيع (هوْل يوم القيامة)
( ۷ ) خشّعا أبصارُهمْ :ذليلة خاضِعة من شدة الهوْل
( ۷ ) الأجْدَاث :القبور
( ۸ ) مُهْطعين :مسْرعين ، مادِّي أعناقهمْ
( ۸ ) يومٌ عَسِر :صعْب شديدٌ لِعِظم أهْواله
( ۹ ) ازدجـِـر :زُجـِـرَ عن تبليغ رسالته بالسّبّ و غيره
( ۱۰ ) مَـغلوبٌ فانتصِرْ :مَقـْـهورٌ فانـْـتـَـقِم لي منهم
( ۱۱ ) أبْْواب السماء :السّحاب
( ۱۱ ) بماءٍ منهمِر :مُنـْـصبّ بشدّة و غزارة
( ۱۲ ) فجّرنا الأرض :شققناها
( ۱۲ ) أمْــٍر قدْ قـُـدِرَ :قدّرْناه أزَلاً (هلاكهم بالطوفان)
( ۱۳ ) دُسُــٍر :مَسَاميرَ تـُـشدّ بها الألواح
( ۱٤ ) تجْري بأعْـيُـننا :بـِـحِـفـْـظِنا أو بمرأى منّا أو بأمْرنا
( ۱۵ ) تـَركْناها آية :أبْقـينا ذكْرَها عِـبرة و عِـظة
( ۱۵ ) مُـدّكِر :مُعتبر، متـّـعِظٍ بها
( ۱٦ ) نُـذر :إنذاري
( ۱۹ ) ريحًا صَرْصَرا :شديدة السّموم أو البرْدأو الصّوت
( ۱۹ ) يَوْم نحْس :شؤْم عليهم
( ۱۹ ) مستمرّ :دائم نحسُهُ. أو مُحكمٍ . أو بَشِع
( ۲۰ ) تنزع النّاس :تقـلعهم منْ أماكنهم و ترمي بهم
( ۲۰ ) أعجاز نخل :أصوله بلا رُءوس
( ۲۰ ) منقعِر :منـْـقلع عن قعْره و مَغرسِه
( ۲٤ ) سُعُر :شدّة عذاب و نار أو جنون
( ۲۵ ) كذاب أشر :بَطِرٌ مُـتكبّرْ
( ۲۷ ) فتنة لهم ْ :امتحانا و ابتلاءً لهم
( ۲۷ ) اصْـطبـِـرْ :اصبرْ على أذاهمْ و لا تعجَلْ
( ۲۸ ) قِـسْمَة بينهم :مَقـسومٌ بَـيـْـنهم و بين النـّـاقة
( ۲۸ ) كلّ شِرْب :كلّ نصيب و حصّة من الماء
( ۲۸ ) مُحْـتـَـضرٌ :يحْضره صاحبه في نـَوبته
( ۲۹ ) فَتـَـعاطى :فتناول النّـاقة بسيْـفِه اجْتراءً منه
( ۳۱ ) كهشيم :كاليابس المُـتـفـتـّـت من شجر الحظيرة
( ۳۱ ) المُحتظِر :صانع الحظيرة (الزّريبة) لمواشيه من هذا الشجر
( ۳٤ ) حاصبًا :ريحًا ترميهم بالحصباء
( ۳٤ ) نجّـيناهم بسَحَر :عـند اصداع الفجْر
( ۳٦ ) أنذرهم بَطْـشتنا :أخْذتـَـنـَـا الشديدة بالعذاب
( ۳٦ ) فتمارَوْا بالنّـذر :فكَذبوا بها مُتشاكّين
( ۳۷ ) راوَدوه عنْ ضيـْـفه :طَلـَـبوا منه تمكينهم منهم
( ۳۷ ) فطمسْنا أعينهم :أعميْناهم أو أزلنا أثرَها بمَسْحِها
( ۳۸ ) بكرةً :أوّل النهار
( ٤۳ ) في الزّبر :في الكتب السّماويّة
( ٤٤ ) نحن جميعٌ :جماعة ، مجتمعٌ أمْرنا
( ٤٤ ) مُنتصر :مُـمْـتـَـنِعٌ ، لا نُغلب
( ٤٦ ) السّاعة أدْهى :أعظم داهية و أفظع
( ٤٦ ) أمَرّ :أشدّ مرارة منْ عذاب الدنيا
( ٤۷ ) سُعُر :نيران مسعّرة أو جُـنـُـون
( ٤۹ ) خلقـناه بقـَـدَر :بتقدير سابق أو مُـقدّرًا مُحْكَما
( ۵۰ ) إلا واحدة :كلمة واحدة ، هي "كُـنْ
( ۵۱ ) أشياعكمْ :أمثالهم في الكفر
( ۵۲ ) الزبر :كتب الحفظة
( ۵۳ ) مستطَر :مسطور مكتوب في اللوح المحفوظ
( ۵٤ ) نـهَر :أنهار
( ۵۵ ) مقعد صدْق :مَكانٍ مَرْضى
