سورة التكوير ﴿ ۲۹ ﴾
بِسمِ اللَّهِ الرَّحمٰنِ الرَّحيمِ
إِذَا الشَّمسُ كُوِّرَت ﴿۱﴾
وَإِذَا النُّجومُ انكَدَرَت ﴿۲﴾
وَإِذَا الجِبالُ سُيِّرَت ﴿۳﴾
وَإِذَا العِشارُ عُطِّلَت ﴿٤﴾
وَإِذَا الوُحوشُ حُشِرَت ﴿۵﴾
وَإِذَا البِحارُ سُجِّرَت ﴿٦﴾
وَإِذَا النُّفوسُ زُوِّجَت ﴿۷﴾
وَإِذَا المَوءۥدَةُ سُئِلَت ﴿۸﴾
بِأَىِّ ذَنبٍ قُتِلَت ﴿۹﴾
وَإِذَا الصُّحُفُ نُشِرَت ﴿۱۰﴾
وَإِذَا السَّماءُ كُشِطَت ﴿۱۱﴾
وَإِذَا الجَحيمُ سُعِّرَت ﴿۱۲﴾
وَإِذَا الجَنَّةُ أُزلِفَت ﴿۱۳﴾
عَلِمَت نَفسٌ ما أَحضَرَت ﴿۱٤﴾
فَلا أُقسِمُ بِالخُنَّسِ ﴿۱۵﴾
الجَوارِ الكُنَّسِ ﴿۱٦﴾
وَالَّيلِ إِذا عَسعَسَ ﴿۱۷﴾
وَالصُّبحِ إِذا تَنَفَّسَ ﴿۱۸﴾
إِنَّهُ لَقَولُ رَسولٍ كَريمٍ ﴿۱۹﴾
ذى قُوَّةٍ عِندَ ذِى العَرشِ مَكينٍ ﴿۲۰﴾
مُطاعٍ ثَمَّ أَمينٍ ﴿۲۱﴾
وَما صاحِبُكُم بِمَجنونٍ ﴿۲۲﴾
وَلَقَد رَءاهُ بِالأُفُقِ المُبينِ ﴿۲۳﴾
وَما هُوَ عَلَى الغَيبِ بِضَنينٍ ﴿۲٤﴾
وَما هُوَ بِقَولِ شَيطٰنٍ رَجيمٍ ﴿۲۵﴾
فَأَينَ تَذهَبونَ ﴿۲٦﴾
إِن هُوَ إِلّا ذِكرٌ لِلعٰلَمينَ ﴿۲۷﴾
لِمَن شاءَ مِنكُم أَن يَستَقيمَ ﴿۲۸﴾
وَما تَشاءونَ إِلّا أَن يَشاءَ اللَّهُ رَبُّ العٰلَمينَ ﴿۲۹﴾
